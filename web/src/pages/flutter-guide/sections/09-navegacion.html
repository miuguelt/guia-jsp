4. Navegación con GoRouter

Rutas declarativas, redirecciones y protección de rutas

Configuración de GoRouter

GoRouter reemplaza al Navigator tradicional con un enfoque declarativo.

router.dart 
import 'package:go_router/go_router.dart';

final routerProvider = Provider<GoRouter>((ref) {
  final authState = ref.watch(authProvider);

  return GoRouter(
    initialLocation: '/productos',
    redirect: (context, state) {
      final estaLogueado = authState.estaAutenticado;
      final login = state.matchedLocation == '/login';

      if (!estaLogueado && !login) return '/login';
      if (estaLogueado && login) return '/productos';
      return null;
    },
    routes: [
      GoRoute(
        path: '/login',
        builder: (_, __) => LoginScreen(),
      ),
      GoRoute(
        path: '/productos',
        builder: (_, __) => ProductosScreen(),
        routes: [
          GoRoute(
            path: ':id',
            builder: (_, state) =>
              ProductoDetalleScreen(
                id: int.parse(state.pathParameters['id']!),
              ),
          ),
        ],
      ),
    ],
  );
});

Rutas Protegidas

El redirect de GoRouter evalúa cada navegación y redirige según el estado de autenticación.

auth_guard.dart 
// Provider de autenticación
@riverpod
class Auth extends _$Auth {
  @override
  AuthState build() => AuthState.noAutenticado();

  Future<void> login(String user, String pass) async {
    state = const AuthState.cargando();
    try {
      final token = await api.login(user, pass);
      state = AuthState.autenticado(token);
    } catch (e) {
      state = AuthState.error(e.toString());
    }
  }

  void logout() => state = AuthState.noAutenticado();
}

// Uso en widgets
class LoginScreen extends ConsumerWidget {
  @override
  Widget build(BuildContext context, WidgetRef ref) {
    return Scaffold(
      body: Center(
        child: FilledButton(
          onPressed: () => ref.read(authProvider.notifier).login(
            'admin', '1234',
          ),
          child: Text('Iniciar sesión'),
        ),
      ),
    );
  }
}
